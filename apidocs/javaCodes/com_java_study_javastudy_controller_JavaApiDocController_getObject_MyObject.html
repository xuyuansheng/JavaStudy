public class MyObject implements Serializable{

    private String name; //
    private int intp; //
    private long longp; //
    private int nameI; //
    private Object[] list; //

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="intp")
    public int getIntp(){
        return intp;
    }

    @JSONField(name="intp")
    public void setIntp(int intp){
        this.intp = intp;
    }

    @JSONField(name="longp")
    public long getLongp(){
        return longp;
    }

    @JSONField(name="longp")
    public void setLongp(long longp){
        this.longp = longp;
    }

    @JSONField(name="nameI")
    public int getNameI(){
        return nameI;
    }

    @JSONField(name="nameI")
    public void setNameI(int nameI){
        this.nameI = nameI;
    }

    @JSONField(name="list")
    public Object[] getList(){
        return list;
    }

    @JSONField(name="list")
    public void setList(Object[] list){
        this.list = list;
    }

}
